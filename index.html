var initialIndex = 0;
var lastIndex = array.length - 1;

while (true && initialIndex <= lastIndex) {
    //as the array is sorted 
    var median = Math.floor((initialIndex + lastIndex) / 2);
    if (array[median] === value) {
        return median
    }

    if (array[median] < value) {

        median++;
        initialIndex = median;

    } else {
        median--;
        lastIndex = median;

    }

}
return -1;
