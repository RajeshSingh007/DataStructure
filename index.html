function areThereDuplicates(...args) {


    
    args.sort((a, b) => a > b);
    let start = 0;
    let next = 1;
    while (next < args.length) {
        if (args[start] === args[next]) {
            return true
        }
        start++
        next++
    }
    return false


}

function countUniqueValues(sortedArray = [1, 1, 2]) {
    

    var i = 0;
    var j = i + 1;

    while (j !== sortedArray.length) {

        if (sortedArray[j] - sortedArray[i] === 0) { // means they are same
            j++;
        } else if (sortedArray[j] - sortedArray[i] > 0) {
            i++;
            sortedArray[i] = sortedArray[j];
            j++;
        }

function isPalindrome(s) {


function recursive(string) {
    

    var newString = string;
    if (newString.length === 0) {
        return "";
    }
    var lastValue = newString.substr(-1);
    var trimmedString = newString.substr(0, (newString.length - 1));

    return lastValue + recursive(trimmedString);
    

}
var finallTrimmedValue = recursive(s);
if (finallTrimmedValue === s) {
    return true;
}
return false;


}
